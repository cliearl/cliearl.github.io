class Tree {
    var root: Node?

    func put(node: Node) {
        if self.root == nil {
            // ë£¨íŠ¸ê°€ ë„ì´ë©´ ê°’ì„ ë°”ë¡œ ì €ì¥
            self.root = node
        } else {
            var c_node = self.root
            var is_put = false
            
            while !is_put {
                if node.key < c_node!.key {
                    if c_node!.l_child == nil {
                        c_node!.l_child = node
                        is_put = true
                    } else {
                        c_node = c_node!.l_child
                    }
                } else {
                    if c_node!.r_child == nil {
                        c_node!.r_child = node
                        is_put = true
                    } else {
                        c_node = c_node!.r_child
                    }
                }
            }
        }
    }

    // ì „ìœ„ ìˆœíšŒ
    func preorder() -> [Int] {
        
        var stack = [self.root]
        var visit = [Int]()

        while !stack.isEmpty {
            // ë§¨ ë’¤ì˜ ë…¸ë“œê°’ì„ ë°˜í™˜ í›„ ì‚­ì œ
            let node = stack.popLast()!
            visit.append(node!.value)
            
            // ì™¼ìª½ ë…¸ë“œê°€ ë§¨ ë’¤ë¡œ ì˜¤ë„ë¡ í•œë‹¤
            if node?.r_child != nil {
                stack.append(node?.r_child)
            }
            if node?.l_child != nil {
                stack.append(node?.l_child)
            }
        }
        
        return visit
    }

    // í›„ìœ„ ìˆœíšŒ
    func postorder() -> [Int] {

        var stack = [self.root]
        var visit = [Int]()
        
        while !stack.isEmpty {
            let node = stack.popLast()!
            visit.append(node!.value)

            // ì˜¤ë¥¸ìª½ ë…¸ë“œê°€ ë§¨ ë’¤ë¡œ ì˜¤ë„ë¡ í•œë‹¤
            if node?.l_child != nil {
                stack.append(node?.l_child)
            }
            if node?.r_child != nil {
                stack.append(node?.r_child)
            }
        }
        
        return visit.reversed()
    }
}

func parse_html(word: String, html: String) -> (String, Int, [String]) {

    func count(target: String, word_list: [String]) -> Int {
        var result = 0

                result += 1
            }
        }

    }
    
    // head -> meta íƒœê·¸ì•ˆì˜ urlì„ ì¶”ì¶œ

    // ê¸°ë³¸ì ìˆ˜ ê³„ì‚°

    return (url[0], normal, external_links)

        if isXYsafe(nx, ny, n) {

func solution(_ board:[[Int]]) -> Int {
    
    n = board.count
    var answer = 0
    var possi = Bool()

    // ì…ë ¥ë°›ëŠ” í–‰ë ¬, ë¸”ë¡ ì²´í¬ í›„ ì§€ì›Œì§€ëŠ”ê²ƒë„ ë°˜ì˜

    //  ê²€ì€ ë¸”ë¡ì„ ë‚´ë¦´ ìˆ˜ ìˆëŠ”ì§€ íŒì •í•˜ëŠ” ë³´ë“œ
    var count = Array(repeating: Array(repeating: 0, count: n), count: n)
    // ì§€ì›Œì•¼ í•  ë¸”ë¡ì„ ì•Œë ¤ì£¼ëŠ” í–‰ë ¬
    var check = Array(repeating: Array(repeating: 0, count: n), count: n)
    
    while true {
        // count, checkë³´ë“œ ì´ˆê¸°í™•
        for i in 0 ..< n {
            for j in 0 ..< n {
                count[i][j] = 0
                check[i][j] = 0
            }
        }
        var flag = false
        
        // board ì •ë³´ë¡œë¶€í„° count ë³´ë“œë¥¼ ì‘ì„±. ë¸”ë¡ì´ ìˆëŠ”ì§€ë§Œ ìš°ì„  íŒì •
        for i in 0 ..< n {
            for j in 0 ..< n {
                if board[i][j] > 0 {
                    count[i][j] = 1
                } else {
                    count[i][j] = 0
                }
            }
        }
        // ë¸”ë¡ ê°’ì„ ë°‘ìœ¼ë¡œ ë‚´ë¦¬ë©´ì„œ ê³„ì† ë”í•œë‹¤. ê·¸ëŸ¼ ê²€ì€ ë¸”ë¡ì€ ë¸”ë¡ ë°‘ì— ì˜¬ ìˆ˜ ì—†ìŒ
        for i in 1 ..< n {
            for j in 0 ..< n {
                count[i][j] += count[i-1][j]
            }
        }
        
        for i in 0 ..< n {
            for j in 0 ..< n {
                for k in 0 ..< 2 {
                    (count, board, check, possi) = possible(i, j, k, count, board, check, n)
                    
                    if possi {
                        // ì—°ì‚°ì´ ëë‚˜ì§€ ì•Šì•˜ìœ¼ë¯€ë¡œ ë£¨í”„ë¥¼ ë‹¤ì‹œ ì‹¤í–‰ì‹œí‚´
                        flag = true

        for i in 0 ..< n {
            for j in 0 ..< n {
                if check[i][j] == 1 {
                    board[i][j] = 0
                }
            }
        }
        
        if !flag {
            break
        }
    }
    
    return answer
}
