import Foundation

func solution(_ board:[[Int]]) -> Int
{
    var newBoard = board
    
    let row = newBoard.count
    let col = newBoard[0].count
    var answer = 0
    
    // ÌñâÏù¥ÎÇò Ïó¥Ïùò Í∏∏Ïù¥Í∞Ä 2 ÎØ∏ÎßåÏù¥ÎùºÎ©¥ ÏßÅÏ†ë ÎèåÎ¶¨Î©¥ÏÑú 1Ïù¥ ÌïòÎÇòÎùºÎèÑ ÏûàÎäîÏßÄ Ï≤¥ÌÅ¨
    // ÌïòÎÇòÎùºÎèÑ ÏûàÏúºÎ©¥ ÌÜµÍ≥º.
    if (col < 2 || row < 2) {
        for i in 0 ..< row {
            for j in 0 ..< col {
                if newBoard[i][j] == 1 {
                    answer = 1
                }
            }
        }
    } else {
        // Î∞∞Ïó¥Ïùò [1][1]Î∂ÄÌÑ∞ Î∞òÎ≥µÎ¨∏ÏùÑ ÎèåÎ¶∞Îã§. (Ï≤´ Î≤àÏß∏ Ìñâ, Ï≤´ Î≤àÏß∏ Ïó¥ Î¨¥Ïãú)
        // ÌòÑÏû¨ Í∞íÏù¥ 1Ïùº Í≤ΩÏö∞, Ï¢åÏ∏°Í∞í, ÏÉÅÎã®Í∞í, Ï¢åÏ∏°ÏÉÅÎã®Í∞í Ï§ë
        // Í∞ÄÏû• ÏûëÏùÄ Í∞íÏùò +1 Ìïú Í∞íÏùÑ ÌòÑÏû¨ Í∞íÏúºÎ°ú Ìï†Îãπ
        for i in 1 ..< row {
            for j in 1 ..< col {
                if newBoard[i][j] == 1 {
                    newBoard[i][j] = min(newBoard[i - 1][j], newBoard[i][j - 1], newBoard[i - 1][j - 1]) + 1
                    if newBoard[i][j] > answer {
                        answer = newBoard[i][j]
                    }
                }
            }
        }
    }
    return answer  - answer
}

import Foundation

func solution(_ land:[[Int]]) -> Int{
    var answer = 0
    var newLand = land
    
    for i in 1 ..< newLand.count {

        newLand[i][0] += max(newLand[i - 1][1], newLand[i - 1][2], newLand[i - 1][3])
        newLand[i][1] += max(newLand[i - 1][0], newLand[i - 1][2], newLand[i - 1][3])
        newLand[i][2] += max(newLand[i - 1][0], newLand[i - 1][1], newLand[i - 1][3])
        newLand[i][3] += max(newLand[i - 1][0], newLand[i - 1][1], newLand[i - 1][2])
    }
    answer = (newLand.last?.max())!
    
    return answer
}

func solution(_ sticker:[Int]) -> Int{

    if sticker.count == 0 { return 0 }

    var dp1 = [Int]()    // Ï≤´Î≤àÏß∏ ÎúØÏùÄ Í≤ΩÏö∞
    var dp2 = [Int]()    // Ï≤´Î≤àÏß∏ ÏïàÎúØÏùÄ Í≤ΩÏö∞

func solution(_ strs:[String], _ t:String) -> Int {
    
    let INF = 9999
    let wordLength = t.count

    // Í≥†ÏÜçÌÉêÏÉâÏùÑ ÏúÑÌïú Ìï¥ÏãúÏÖã ÏÉùÏÑ±

    let splitT = t.map { String($0) }

    // dp[i] = min(dp[i+1],dp[i+2],dp[i+3],dp[i+4]) + 1
    var dp = [Int](repeating: INF, count: wordLength)
    dp.append(0)

    
    // Í∞ÅÏûêÎ¶¨Î≥ÑÎ°ú ÏÇ¨Ïö© Í∞ÄÎä•Ìïú Îã®Ïñ¥Ï°∞Í∞ÅÏùò ÏµúÏÜåÍ∞íÏùÑ Îí∑ÏûêÎ¶¨Î∂ÄÌÑ∞ Í≥ÑÏÇ∞
    for i in (0 ..< wordLength).reversed() {
        // Îã®Ïñ¥Ï°∞Í∞Å ÏµúÎåÄ Í∏∏Ïù¥Îäî 5
        for j in 0 ..< 5 {
            let k = i + j + 1
            // Í≤ÄÏÉâ Ïù∏Îç±Ïä§Î•º ÎÑòÏñ¥Í∞ÄÎ©¥ Ï¢ÖÎ£å
            if k > wordLength {
                break
            }
            
            // ÏÇ¨Ïö©Í∞ÄÎä•Ìïú Îã®Ïñ¥Í∞Ä ÏûàÎäîÏßÄ ÌôïÏù∏
            // SwiftÏùò String.indexÎäî Ïì∞Í∏∞Îäî Ìé∏ÌïòÏßÄÎßå Îß§Ïö∞ ÎäêÎ¶¨Îã§
            // let availWord = String(t[t.index(t.startIndex, offsetBy: i)..<t.index(t.startIndex, offsetBy: k)])
            // Í∑∏ÎûòÏÑú Îã®Ïñ¥Îß§ÏπòÎ•º Ìï†Îïê Î∞∞Ïó¥Î°ú ÏßÅÏ†ë Í∞ÄÏ†∏ÏôÄÏÑú Ìï¥ÏãúÏÖãÍ≥º ÎπÑÍµêÌï¥Ïïº ÎêúÎã§

            var availWord = String()

            // Ï°∞Ìï©Ïù¥ Í∞ÄÎä•ÌïòÎ©¥ dp[i]Ïùò Í∞íÏùÑ ÏµúÏÜåÍ∞íÏúºÎ°ú Í∞±Ïã†
            if (strsSet.contains(availWord) && dp[i + j + 1] != INF) {
